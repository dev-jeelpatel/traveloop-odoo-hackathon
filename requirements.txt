# ENVIRONMENT VARIABLES  (backend/.env)
# ─────────────────────────────────────────────────────────────────────────────

# ─────────────────────────────────────────────────────────────────────────────
# QUICK SETUP COMMANDS
# ─────────────────────────────────────────────────────────────────────────────
# 1. Backend:
#    cd backend
#    npm install
#    cp .env.example .env        # edit .env with your DB + email values
#    npx prisma db push          # creates DB schema
#    node prisma/seed.js         # seeds cities + activities
#    npm run dev                 # → http://localhost:5000
#
# 2. Frontend (User App):
#    cd frontend
#    npm install
#    npm run dev                 # → http://localhost:5173
#
# 3. Admin Panel:
#    cd admin
#    npm install
#    npm run dev                 # → http://localhost:5174/admin/login
